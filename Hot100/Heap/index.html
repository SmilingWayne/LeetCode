import heapq
class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        record = defaultdict(int)
        # heappush
        # heapppushpop
        for num in nums:
            record[num] += 1

        array = []
        for val, cnt in record.items():
            if len(array) < k:
                heapq.heappush(array, (cnt, val))
            elif  cnt > array[0][0]:
                heapq.heappushpop(array, (cnt, val))

        result = []
        for val in array:
            result.append(val[1])
        return result
